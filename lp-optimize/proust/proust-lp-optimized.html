<script>
(function () {
  const LIMIT_MIN = 20;
  const PREFIX      = "offer_timer_proust-cream2_v7-5";
  const REMAIN_KEY  = PREFIX + "remainSec";
  const EXPIRE_URL = "https://shop.rmh.co.jp/lp?u=me_proust2_cartsingle";

  if (location.search.includes("reset=1")) { localStorage.removeItem(REMAIN_KEY); }

  const DEFAULT_SEC = LIMIT_MIN * 60;
  let remainSec = parseInt(localStorage.getItem(REMAIN_KEY), 10);
  if (isNaN(remainSec) || remainSec < 0 || remainSec > DEFAULT_SEC) {
    remainSec = DEFAULT_SEC;
    localStorage.setItem(REMAIN_KEY, String(remainSec));
  }

  let timerId = null;

  function render() {
    const lpTimers = document.querySelectorAll(".lp-timer, #lp-timer");
    const chatTimer = document.getElementById("chat-header-timer");
    const sec = Math.max(remainSec, 0);
    const m = String(Math.floor(sec / 60)).padStart(2, "0");
    const s = String(sec % 60).padStart(2, "0");
    const text = `残り時間 ${m}分${s}秒`;
    lpTimers.forEach(el => { el.textContent = text; });
    if (chatTimer) chatTimer.textContent = text;
  }

  function stopTick() { if (timerId !== null) { clearInterval(timerId); timerId = null; } }

  function handleExpire() {
    remainSec = 0;
    localStorage.setItem(REMAIN_KEY, "0");
    render(); stopTick();
    window.location.href = EXPIRE_URL;
  }

  function startTick() {

    if (document.hidden) { stopTick(); } else { startTick(); }
  });
