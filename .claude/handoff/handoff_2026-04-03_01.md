```python
def clean_telop_text(text: str) -> str:
    """テロップ用テキストクリーニング。句読点・引用符を除去。"""
    # 削除対象
    text = text.replace("、", "").replace("。", "")
    text = text.replace("「", "").replace("」", "")
    # …！？は残す（感情・余韻の演出意図）
    # ※以降は別処理（既存のmake_telopで小フォント表示）
    return text
```
